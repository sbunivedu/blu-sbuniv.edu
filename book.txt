CREATE TABLE book(
  book_id INTEGER,
  book_name TEXT,
  price REAL,
  quantity INTEGER,
  pub_id INTEGER
);

CREATE TABLE publisher(
  pub_id INTEGER,
  pub_name TEXT
);

CREATE TABLE book_log(
  book_id INTEGER,
  operation TEXT,
  old_quantity INTEGER,
  new_quantity INTEGER,
  old_price REAL,
  new_price REAL
);
